[docs]    def density(self, Tk=None, Tc=None):
        """
        Return the density of Sodium Chloride.

        Notes
        -----
        From equation 10 of Thermophysical Properties of NaCl
        NaBr and NaF by y-ray attenuation technique
        """
        Tk = getTk(Tc, Tk)
        return -3.130e-04 * Tk + 2.23
