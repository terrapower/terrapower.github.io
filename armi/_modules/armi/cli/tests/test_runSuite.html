    def test_listCommand(self):
        """Ensure run-suite entry point is registered.

        .. test:: The ARMI CLI can be correctly initialized.
            :id: T_ARMI_CLI_CS0
            :tests: R_ARMI_CLI_CS
        """
        acli = ArmiCLI()

        origout = sys.stdout
        try:
            out = io.StringIO()
            sys.stdout = out
            acli.listCommands()
        finally:
            sys.stdout = origout

        self.assertIn("run-suite", out.getvalue())
